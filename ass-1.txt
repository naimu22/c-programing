#include<stdio.h>
void main()
{
	char s[20];
	int i,x=0,cnt=0,y=0;
	gets(s);
	for(i=0;s[i]!='\0';i++)
	{
		if(isupper(s[i]))
			x=x+1;
		else if(islower(s[i]))
			cnt=cnt+1;
		else if(isdigit(s[i]))
			y=y+1;
	}
	printf("\nCount of uppercase letters: %d",x);
	printf("\nCount of lowercase letters: %d",cnt);
	printf("\nCount of numbers: %d",y);
}
